const handleWifiClick = () => {
  if (!hasVisitedInstagram) {
    // Guardar que ya visito Instagram
    localStorage.setItem('hasVisitedInstagram', 'true')
    setHasVisitedInstagram(true)

    // Redirigir a Instagram directamente (evita about:blank)
    window.location.href = instagramUrl

    // NO mostrar la contrasena inmediatamente
    // El usuario volvera y entonces la vera
  } else {
    // Si ya visito Instagram, mostrar la contrasena con delay de 2 segundos
    setTimeout(() => {
      setShowWifiPassword(true)
    }, 2000)
  }
}
